2 MATH Library
2.34 EXP

EXP(X,PREC) ;
	New L,LIM,K,VALUE
	Set PREC=$Get(PREC,11)
	Set L=X,VALUE=X+1
	Set LIM=$Select((PREC+3)'>11:PREC+3,1:11),@("LIM=1E-"_LIM)
	For K=2:1 Set L=L*X/K,VALUE=VALUE+L Quit:($Translate(L,"-")<LIM)
	Quit VALUE
