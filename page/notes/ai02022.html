2 MATH Library
2.22 COSH

COSH(X,PREC) ;
	New E,F,I,P,R,T,XX
	Set PREC=$Get(PREC,11)+1
	Set @("E=1E-"_PREC)
	Set XX=X*X,F=1,(P,R,T)=1,I=1
	For  Set T=T*XX,F=I+1*I*F,R=T/F+R,P=P-R/R,I=I+2 If -E<P,P<E Quit
	Quit R
