2 MATH Library
2.47 MTXUNIT

MTXUNIT(R,N,SPARSE) ;
	; Create a unit matrix R[N,N]
	If $Get(N)<1 Quit 0
	;
	New ROW,COL
	For ROW=1:1:N For COL=1:1:N Do
	. KValue R(ROW,COL)
	. If $Get(SPARSE) Quit:ROW'=COL
	. Set R(ROW,COL)=$Select(ROW=COL:1,1:0)
	. Quit
	Quit 1
