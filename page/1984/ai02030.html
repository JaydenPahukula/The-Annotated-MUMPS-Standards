2 MATH Library
2.30 DECDMS

DECDMS(X,PREC) New T
	Set PREC=$Get(PREC,5)
	Set X=X#360*3600
	Set T=PREC-$Length(X\1)
	Set X=+$Justify(X,0,$Select(T'<0:T,1:0))
	Quit X\3600_":"_(X\60#60)_":"_(X#60)
