2 MATH Library
2.24 COTH

COTH(X,PREC) ;
	New SINH
	If 'X Quit "INFINITE"
	;
	Set PREC=$Get(PREC,11)
	Set SINH=$%SINH^MATH(X,PREC)
	If 'SINH Quit "INFINITE"
	Quit $%COSH^MATH(X,PREC)/SINH
