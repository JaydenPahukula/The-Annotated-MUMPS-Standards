2 MATH Library
2.43 MTXMUL

MTXMUL(A,B,R,M,L,N) ;
	; Multiply A[M,L] by B[L,N], result goes to R[M,N]
	If $Data(A)<10 Quit 0
	If $Data(B)<10 Quit 0
	If $Get(L)<1 Quit 0
	If $Get(M)<1 Quit 0
	If $Get(N)<1 Quit 0
	;
	New I,J,K,SUM,ANY
	For I=1:1:M For J=1:1:N Do
	. Set (SUM,ANY)=0
	. KValue R(I,J)
	. For K=1:1:L Do
	. . Set:$Data(A(I,K))#2 ANY=1
	. . Set:$Data(B(K,J))#2 ANY=1
	. . Set SUM=$Get(A(I,K))*$Get(B(K,J))+SUM
	. . Quit
	. Set:ANY R(I,J)=SUM
	. Quit
	Quit 1
