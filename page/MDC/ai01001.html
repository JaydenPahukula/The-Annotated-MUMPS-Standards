1 CHARACTER Library
1.1 COLLATE

COLLATE(A,CHARMOD) New x
	Set x=""
	If $Get(CHARMOD)'="" Do
	. If $Extract(CHARMOD,1)="^" Do
	. . Set x=$Extract(CHARMOD,2,$Length(CHARMOD))
	. . If x'="" Set x=$Get(^$Global(x,"COLLATE"))
	. . Quit
	. If x="" Set x=$Get(^$Character(CHARMOD,"COLLATE"))
	. Quit
	If x="" Set x=^$Job($Job,"COLLATE")
	Set x=@(x_"("_A_")")
	Quit x
