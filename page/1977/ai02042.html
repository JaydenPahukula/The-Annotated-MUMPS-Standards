2 MATH Library
2.42 MTXINV

MTXINV(A,R,N) ;
	; Invert A[N,N], result goes to R[N,N]
	If $Data(A)<10 Quit 0
	If $Get(N)<1 Quit 0
	;
	New T,X
	Set X=$%MTXUNIT^MATH(.T,N)
	Quit $%MTXEQU^MATH(.A,.T,.R,N,N)
